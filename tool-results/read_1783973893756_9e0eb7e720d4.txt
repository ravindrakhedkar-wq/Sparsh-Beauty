   246→export default function AcademySection() {
   247→  const ref = useRef(null)
   248→  const isInView = useInView(ref, { once: true, margin: '-100px' })
   249→  const [courses, setCourses] = useState<Course[]>([])
   250→  const [loading, setLoading] = useState(true)
   251→  const [enquiryOpen, setEnquiryOpen] = useState(false)
   252→  const [detailOpen, setDetailOpen] = useState(false)
   253→  const [selectedCourse, setSelectedCourse] = useState<Course | null>(null)
   254→  const [enquiryForm, setEnquiryForm] = useState({ name: '', mobile: '', email: '', course: '' })
   255→  const [submitting, setSubmitting] = useState(false)
   256→
   257→  useEffect(() => {
   258→    async function fetchCourses() {
   259→      try {
   260→        const res = await fetch('/api/courses')
   261→        if (res.ok) setCourses(await res.json())
   262→      } catch { /* silent */ }
   263→      finally { setLoading(false) }
   264→    }
   265→    fetchCourses()
   266→  }, [])
   267→
   268→  const openEnquiry = (courseTitle: string) => {
   269→    setSelectedCourse(courses.find(c => c.title === courseTitle) || null)
   270→    setEnquiryForm(prev => ({ ...prev, course: courseTitle }))
   271→    setEnquiryOpen(true)
   272→  }
   273→
   274→  const openDetail = (course: Course) => {
   275→    setSelectedCourse(course)
   276→    setDetailOpen(true)
   277→  }
   278→
   279→  const submitEnquiry = async () => {
   280→    if (!enquiryForm.name || !enquiryForm.mobile) {
   281→      toast.error('Please fill in your name and mobile number')
   282→      return
   283→    }
   284→    setSubmitting(true)
   285→    try {
   286→      const res = await fetch('/api/contact', {
   287→        method: 'POST',
   288→        headers: { 'Content-Type': 'application/json' },
   289→        body: JSON.stringify({
   290→          name: enquiryForm.name, mobile: enquiryForm.mobile,
   291→          email: enquiryForm.email || undefined,
   292→          subject: `Course Enquiry: ${selectedCourse?.title}`,
   293→          message: `I am interested in the ${selectedCourse?.title} course. Please provide more details.`,
   294→        }),
   295→      })
   296→      if (res.ok) {
   297→        toast.success('Enquiry submitted!', { description: 'We will get back to you soon.' })
   298→        setEnquiryOpen(false)
   299→        setEnquiryForm({ name: '', mobile: '', email: '', course: '' })
   300→      }
   301→    } catch { toast.error('Failed to submit enquiry') }
   302→    finally { setSubmitting(false) }
   303→  }
   304→
   305→  const handleDownloadBrochure = () => {
   306→    const link = document.createElement('a')
   307→    link.href = '/Sparsh-Academy-Brochure.pdf'
   308→    link.download = 'Sparsh-Beauty-Academy-Brochure.pdf'
   309→    document.body.appendChild(link)
   310→    link.click()
   311→    document.body.removeChild(link)
   312→    toast.success('Brochure download started!')
   313→  }
   314→
   315→  const extra = selectedCourse ? courseExtras[selectedCourse.title] : null
   316→
   317→  return (
   318→    <section id="academy" className="section-padding relative overflow-hidden" ref={ref}>

   321→        <div className="absolute inset-0 bg-cream/95" />
   322→      </div>
   323→
   324→      <div className="relative z-10 max-w-6xl mx-auto">
   325→        <motion.div initial={{ opacity: 0, y: 20 }} animate={isInView ? { opacity: 1, y: 0 } : {}} transition={{ duration: 0.6 }} className="text-center mb-10">
   326→          <h2 className="text-2xl md:text-3xl lg:text-4xl font-bold font-[family-name:var(--font-playfair)] text-charcoal mb-3">
   327→            Beauty <span className="text-gold-gradient">Academy</span>
   328→          </h2>
   329→          <div className="luxury-divider mb-4" />
   330→          <p className="text-xl md:text-2xl font-[family-name:var(--font-playfair)] text-charcoal font-medium mb-3">
   331→            Learn. Practice. <span className="text-rose-gold">Grow.</span>
   332→          </p>
   333→          <p className="text-warm-gray font-[family-name:var(--font-lato)] max-w-2xl mx-auto">
   334→            Join our professional beauty training programs and build a successful career in the beauty industry with hands-on experience and expert guidance.
   335→          </p>
   336→        </motion.div>
   337→
   338→        <motion.div initial={{ opacity: 0 }} animate={isInView ? { opacity: 1 } : {}} transition={{ delay: 0.3 }} className="flex justify-center mb-8">
   339→          <Button onClick={handleDownloadBrochure} variant="outline" className="border-gold text-gold hover:bg-gold/10 rounded-full font-[family-name:var(--font-lato)]">
   340→            <Download className="w-4 h-4 mr-2" />

   344→
   345→        {loading ? (

   384→                    {/* Quick info */}
   385→                    <div className="space-y-2 mb-4">

   388→                          <div className="flex items-center gap-2 text-xs text-warm-gray font-[family-name:var(--font-lato)]">

   417→          </div>

   420→
   421→      {/* Course Detail Dialog */}
   422→      <AnimatePresence>
   423→        {selectedCourse && extra && (

   427→                <DialogTitle className="font-[family-name:var(--font-playfair)] text-charcoal text-xl pr-8">

   435→                </div>
   436→              </DialogHeader>
   437→
   438→              <div className="space-y-6 mt-2">
   439→                {/* Schedule & Batch */}
   440→                <div className="grid grid-cols-1 sm:grid-cols-3 gap-3">
   441→                  <div className="bg-cream rounded-xl p-3">
